#include "investment.h"
#include <sstream>
#include <vector>
 
vector<string> split(string data, string delim){
    size_t pos = 0;
    string temp(data);
    string token;
    vector<string> result;
 
    while ((pos = temp.find(delim)) != string::npos) {
        token = temp.substr(0, pos);
        result.push_back(token);
        temp.erase(0, pos + delim.length());
    }
 
    result.push_back(temp);
    return result;
}
 
Investment::Investment(): date(""), high(0), low(0){
}
 
Investment::Investment(string data){
    vector<string> tokens = split(data, ",");
    this->date = tokens[0];
 
    stringstream sstream1(tokens[1]);
    int low;
    sstream1 >> low;
    this->low = low;
 
    stringstream sstream(tokens[2]);
    int high;
    sstream >> high;
    this->high = high;
}

//

//  Created by Shawn Hung on 2020/6/3.
//  Copyright © 2020 Shawn Hung. All rights reserved.
//
 
#include "account.h"
#include <iostream>
 
using namespace std;
 
Account::Account(const int initCash) :
investmentCount(0), cash(initCash), investments(new Investment[50]())
{}
 
Account::~Account() throw(){
    delete [] this->investments;
}
 
void Account::buy(Investment *datalist, size_t dataLength, int index){
    if (index > dataLength - 1) {
        cout << "index out of range" << endl;
        return;
    }
 
    Investment invest = datalist[index];
    this->investments[this->investmentCount++] = invest;
    this->cash -= invest.low;
}
 
void Account::sellLast(const int currentPrice){
    if (this->investmentCount == 0) {
        cout << "no investment" << endl;
        return;
    }
    Investment invest = this->investments[this->investmentCount--];
    this->cash += currentPrice - invest.low;
}
 
void Account::list(){
 
    if (this->investmentCount == 0) {
        cout << "no investment" << endl;
        return;
    }
 
    for (int i=0; i < this->investmentCount; i++) {
        Investment invest = this->investments[i];
        cout << invest.date << " " << invest.low << " " << invest.high<< endl;
    }
}
 
long Account::getCash(){
    return this->cash;
}
